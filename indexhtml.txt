var acTimer=null;
function acSearch(inp,dropId){
  var txt=inp.value; var drop=document.getElementById(dropId);
  clearTimeout(acTimer);
  if(txt.length<2){drop.classList.remove('open');return;}
  acTimer=setTimeout(function(){
    google.script.run
      .withSuccessHandler(function(list){
        if(!list.length){drop.classList.remove('open');return;}
        drop.innerHTML=list.map(function(o){
          return '<div class="ac-item" onmousedown="acPick(\''+dropId+'\',\''+xe(o.dni)+'\',\''+xe(o.nombre)+'\',event)">'
            +'<div class="ac-name">'+xe(o.nombre)+'</div>'
            +'<div class="ac-dni">'+xe(o.dni)+'</div></div>';
        }).join('');
        drop.classList.add('open');
      })
      .withFailureHandler(function(){drop.classList.remove('open');})
      .buscarOperariosPorNombre(txt);
  },250);
}
function acPick(dropId,dni,nombre,evt){
  if(evt)evt.preventDefault();
  var drop=document.getElementById(dropId);
  drop.classList.remove('open');
  // Para la vista individual
  if(dropId==='ac-ind'){
    document.getElementById('inp-nombre').value=nombre;
    document.getElementById('inp-nombre').readOnly=false;
    document.getElementById('disp-dni').textContent=dni;
  }
  // Para la vista Excel: el caller maneja esto via xlAcPick
  if(dropId.startsWith('ac-xl-')){
    var rowId=dropId.replace('ac-xl-','');
    xlAcPick(rowId,dni,nombre);
  }
}

// Navegar con Tab/Enter entre celdas
function xlKey(evt,id,col){
  if(evt.key==='Tab'||evt.key==='Enter'){
    evt.preventDefault();
    var order=['dni','nom','lab','lot','hi','hf'];
    var map={dni:'xl-dni-',nom:'xl-nom-',lab:'xl-lab-',lot:'xl-lot-',hi:'xl-hi-',hf:'xl-hf-'};
    var idx=order.indexOf(col);
    if(idx<order.length-1){
      var next=document.getElementById(map[order[idx+1]]+id);
      if(next)next.focus();
    } else {
      // mover a la siguiente fila, primera celda
      var rowIds=XL_ROWS.map(function(r){return r.id;});
      var ri=rowIds.indexOf(id);
      if(ri<rowIds.length-1){
        var nid=rowIds[ri+1];
        var nxt=document.getElementById('xl-dni-'+nid);
        if(nxt)nxt.focus();
      } else { xlAddRow(); }
    }
  }
}

function xlGuardar(){
  var btn=document.getElementById('btn-xl-save');
  var filas=[];
  var hasData=false;
  document.getElementById('xl-body').querySelectorAll('tr').forEach(function(tr){
    var id=tr.id.replace('xl-tr-','');
    var dni =(document.getElementById('xl-dni-'+id)||{}).value||'';
    var nom =(document.getElementById('xl-nom-'+id)||{}).value||'';
    var est =(document.getElementById('xl-est-'+id)||{}).value||'Trabajo';
    var lab =(document.getElementById('xl-lab-'+id)||{}).value||'';
    var lot =(document.getElementById('xl-lot-'+id)||{}).value||'';
    var hi  =(document.getElementById('xl-hi-' +id)||{}).value||'';
    var hf  =(document.getElementById('xl-hf-' +id)||{}).value||'';
    if(dni.trim()||nom.trim()){
      hasData=true;
      filas.push({dni:dni.trim(),nombre:nom.trim(),estado:est,
                  codigoLabor:lab.trim(),lote:lot.trim(),horaInicio:hi,horaFin:hf});
    }
  });
  if(!hasData){showT('t-xl-er','No hay datos para guardar','er');return;}
  // Validar mínimos
  var errores=[];
  filas.forEach(function(f,i){
    if(!f.dni||f.dni.length!==8)errores.push('Fila '+(i+1)+': DNI inválido');
    else if(!f.nombre)errores.push('Fila '+(i+1)+': nombre vacío');
    else if(f.estado==='Trabajo'&&(!f.codigoLabor||!f.lote))
      errores.push('Fila '+(i+1)+': faltan labor y lote');
  });
  if(errores.length){showT('t-xl-er',errores[0]+' ('+errores.length+' error(es))','er');return;}

  btn.innerHTML='<span class="spin"></span> Guardando '+filas.length+' registros...';
  btn.disabled=true;
  document.getElementById('t-xl-ok').classList.remove('on');
  document.getElementById('t-xl-er').classList.remove('on');

  google.script.run
    .withSuccessHandler(function(r){
      btn.innerHTML='<svg viewBox="0 0 16 16" width="13" height="13" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round" style="margin-right:5px"><polyline points="14 4 6 13 2 9"/></svg>Guardar todo';
      btn.disabled=false;
      if(r.ok){
        showT('t-xl-ok','Se guardaron '+r.guardados+' registro(s) correctamente','ok');
        // Marcar filas guardadas con punto verde
        var idx=0;
        document.getElementById('xl-body').querySelectorAll('tr').forEach(function(tr){
          var id=tr.id.replace('xl-tr-','');
          var dni=(document.getElementById('xl-dni-'+id)||{}).value||'';
          if(dni.trim()&&idx<r.guardados){
            var rn=tr.querySelector('.rn');
            if(rn)rn.innerHTML='<span class="saved-dot"></span>';
            idx++;
          }
        });
      } else showT('t-xl-er',r.mensaje||'Error al guardar','er');
    })
    .withFailureHandler(function(e){
      btn.innerHTML='<svg viewBox="0 0 16 16" width="13" height="13" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round" style="margin-right:5px"><polyline points="14 4 6 13 2 9"/></svg>Guardar todo';
      btn.disabled=false; showT('t-xl-er','Error: '+e.message,'er');
    })
    .registrarMasivo(SUP,filas);
}

var html5QrCode = null;
var currentCameraId = null;
var camerasList = [];
var currentDniMode = 'kp'; // 'kp' o 'qr'

function switchDniMode(mode) {
  currentDniMode = mode;
  document.getElementById('dni-kp-container').style.display = mode === 'kp' ? 'block' : 'none';
  document.getElementById('dni-qr-container').style.display = mode === 'qr' ? 'block' : 'none';
  
  var btnKp = document.getElementById('btn-toggle-kp');
  var btnQr = document.getElementById('btn-toggle-qr');
  
  if (mode === 'kp') {
    btnKp.className = 'btn btn-xs btn-red';
    btnQr.className = 'btn btn-xs btn-ghost';
    stopQrScan();
  } else {
    btnKp.className = 'btn btn-xs btn-ghost';
    btnQr.className = 'btn btn-xs btn-red';
  }
  clrT();
}

function playBip() {
  try {
    var audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    var oscillator = audioCtx.createOscillator();
    var gainNode = audioCtx.createGain();
    
    oscillator.connect(gainNode);
    gainNode.connect(audioCtx.destination);
    
    oscillator.type = 'sine';
    oscillator.frequency.value = 1000; 
    gainNode.gain.setValueAtTime(0.3, audioCtx.currentTime); 
    
    oscillator.start();
    gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.15); 
    oscillator.stop(audioCtx.currentTime + 0.15);
  } catch (e) {
    console.warn("Audio Context no soportado o no permitido", e);
  }
}

function startQrScan() {
  document.getElementById('qr-options-view').style.display = 'none';
  document.getElementById('qr-scanner-view').style.display = 'block';
  document.getElementById('qr-result-box').style.display = 'none';
  clrT();

  if (!html5QrCode) {
    html5QrCode = new Html5Qrcode("qr-reader");
  }

  Html5Qrcode.getCameras().then(function(devices) {
    if (devices && devices.length) {
      camerasList = devices;
      // Por defecto la cámara trasera (suele tener 'back' o 'entorno' en la etiqueta, o es la última)
      var backCam = devices.find(function(device) {
        return device.label.toLowerCase().indexOf('back') > -1 || 
               device.label.toLowerCase().indexOf('entorno') > -1;
      });
      var selectCam = backCam ? backCam : devices[devices.length - 1];
      currentCameraId = selectCam.id;
      
      startScanning(currentCameraId);
    } else {
      startScanning({ facingMode: "environment" });
    }
  }).catch(function(err) {
    startScanning({ facingMode: "environment" });
  });
}

function startScanning(cameraConfig) {
  html5QrCode.start(
    cameraConfig,
    {
      fps: 15,
      qrbox: { width: 200, height: 200 }
    },
    onQrCodeSuccess,
    onQrCodeError
  ).catch(function(err) {
    showT('t-dni', 'No se pudo iniciar la cámara: ' + err, 'er');
    stopQrScan();
  });
}

function stopQrScan() {
  if (html5QrCode && html5QrCode.isScanning) {
    html5QrCode.stop().then(function() {
      document.getElementById('qr-scanner-view').style.display = 'none';
      document.getElementById('qr-options-view').style.display = 'block';
    }).catch(function(err) {
      console.error('Error al detener escáner QR:', err);
    });
  } else {
    document.getElementById('qr-scanner-view').style.display = 'none';
    document.getElementById('qr-options-view').style.display = 'block';
  }
}

function changeCamera() {
  if (camerasList.length < 2) {
    showT('t-dni', 'Solo se detectó una cámara en este dispositivo.', 'er');
    return;
  }
  if (!html5QrCode || !html5QrCode.isScanning) return;

  var currentIndex = camerasList.findIndex(function(c) { return c.id === currentCameraId; });
  var nextIndex = (currentIndex + 1) % camerasList.length;
  currentCameraId = camerasList[nextIndex].id;

  html5QrCode.stop().then(function() {
    startScanning(currentCameraId);
  }).catch(function(err) {
    console.error('Error al cambiar de cámara:', err);
  });
}

function onQrCodeSuccess(decodedText) {
  playBip();
  
  // Limpiar caracteres no numéricos
  var cleanDni = decodedText.replace(/\D/g, '');
  
  // Si es más largo que 8, tomar los últimos 8
  if (cleanDni.length > 8) {
    cleanDni = cleanDni.slice(-8);
  }
  
  // Rellenar ceros a la izquierda
  while (cleanDni.length < 8 && cleanDni.length > 0) {
    cleanDni = '0' + cleanDni;
  }

  if (cleanDni.length !== 8) {
    showT('t-dni', 'Código no contiene un DNI válido (8 dígitos): ' + decodedText, 'er');
    return;
  }

  // Visualizar resultado
  var resBox = document.getElementById('qr-result-box');
  resBox.style.display = 'flex';
  document.getElementById('qr-scanned-val').textContent = cleanDni;

  // Detener escaneo
  if (html5QrCode && html5QrCode.isScanning) {
    html5QrCode.stop().then(function() {
      document.getElementById('qr-scanner-view').style.display = 'none';
      document.getElementById('qr-options-view').style.display = 'block';
      procesarDNI(cleanDni);
    });
  } else {
    procesarDNI(cleanDni);
  }
}

function onQrCodeError(errorMessage) {
  // Ignorar errores de lectura de frames (común mientras busca)
}

function handleQrFile(input) {
  if (input.files.length === 0) return;
  var file = input.files[0];
  clrT();
  
  if (!html5QrCode) {
    html5QrCode = new Html5Qrcode("qr-reader");
  }

  var uploadBtn = document.getElementById('qr-drag-drop');
  var originalHTML = uploadBtn.innerHTML;
  uploadBtn.innerHTML = '<span class="spin spin-r"></span> Leyendo archivo de imagen...';

  html5QrCode.scanFile(file, true)
    .then(function(decodedText) {
      uploadBtn.innerHTML = originalHTML;
      input.value = '';
      onQrCodeSuccess(decodedText);
    })
    .catch(function(err) {
      uploadBtn.innerHTML = originalHTML;
      input.value = '';
      showT('t-dni', 'No se pudo detectar ningún código QR en la imagen.', 'er');
    });
}
